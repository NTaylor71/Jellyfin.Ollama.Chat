async def process_job(job_obj: dict) -> str:
    """
    Process a chat job from the Redis queue.

    Args:
        job_obj (dict): The job payload containing 'query' and 'job_id'.

    Returns:
        str: The formatted result to be sent back to the user.
    """
    query = job_obj.get("query", "")
    user_id = job_obj.get("user_id", "unknown")

    print(f"💬 Received query: {query}")

    # Step 1: Search FAISS (mocked here with a fake search API)
    search_results = await search_faiss(query)

    # Step 2: Format response
    formatted_response = format_results(query, search_results)

    return formatted_response

async def search_faiss(query: str) -> list:
    """
    Perform a search against the FAISS service.

    Args:
        query (str): The user's query.

    Returns:
        list: List of search hits.
    """
    async with httpx.AsyncClient() as client:
        try:
            response = await client.post(f"{VECTORDB_URL}/search", json={"query": query})
            response.raise_for_status()
            return response.json().get("results", [])
        except Exception as e:
            print(f"⚠️ FAISS search failed: {e}")
            return []

def format_results(query: str, results: list) -> str:
    """
    Format FAISS search results into a human-readable string.

    Args:
        query (str): The original user query.
        results (list): List of FAISS search results.

    Returns:
        str: The formatted result string.
    """
    if not results:
        return f"Sorry, no results found for: {query}"

    lines = [f"Results for '{query}':"]
    for idx, result in enumerate(results, 1):
        item = result.get("document", "Unknown item")
        lines.append(f"{idx}. {item}")

    return "\n".join(lines)
